CREATE TABLE Prayer
(
  prayer_id BIGINT       NOT NULL AUTO_INCREMENT COMMENT '기도id',
  userId    BIGINT       NOT NULL COMMENT '사용자id',
  prayer_dt DATE         NOT NULL COMMENT '기도일',
  title     VARCHAR(100) NOT NULL COMMENT '기도전체제목',
  note      TEXT         NULL     COMMENT '노트',
  reg_dt    DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '등록일',
  upd_dt    DATETIME     NULL     DEFAULT  CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일',
  PRIMARY KEY (prayer_id)
) COMMENT '기도';




CREATE TABLE Prayer_topic

 CREATE TABLE bible_version
(
  version_id   VARCHAR(20)  NOT NULL PRIMARY KEY COMMENT '버전아이디: KRV, KJV, NIV',
  version_name VARCHAR(100) NOT NULL COMMENT '버전명: 표준킹제임스성경',

  publisher    VARCHAR(100) NULL     COMMENT '출판사명: 바이블빌리버',

  order_no     SMALLINT     NULL      COMMENT '순서',

 CREATE TABLE bible_book 
(
  book_id       SMALLINT NOT NULL     COMMENT '성경의 책 순서 (1~66)',
  version_id    VARCHAR(20)  NOT NULL     COMMENT '성경 번역 ID (FK)',
  book_code     VARCHAR(20)  NOT NULL     COMMENT '성경 책 코드 (예: GEN)',
  name_ko       VARCHAR(100) NOT NULL     COMMENT '책 이름 (한글)',
  name_en       VARCHAR(100) NOT NULL     COMMENT '책 이름 (영문)',
  title         VARCHAR(100) NULL     COMMENT '책 제목',
  testament     VARCHAR(10)  NOT NULL     COMMENT '신/구약여부: OLDT, NEWT',
  book_abbr     VARCHAR(20)  NOT NULL COMMENT '각 버전별 약어 (예: Gen, Ge, 창 등)',
  chapter_count SMALLINT     NOT NULL     COMMENT '장 수',
  status        TINYINT      NOT NULL DEFAULT 0 COMMENT '0=시작전, 1=진행중, 2=완료',
  upd_dt        TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정일', 
  PRIMARY KEY (book_id, version_id), 
  CONSTRAINT fk_bible_book_version FOREIGN KEY (version_id) REFERENCES bible_version(version_id)
) COMMENT '번역본별 책 정보';


CREATE TABLE bible_verse
(
  verse_id   BIGINT        NOT NULL AUTO_INCREMENT  COMMENT '절 ID',
  version_id VARCHAR(20)   NOT NULL COMMENT '성경 번역 ID (FK)',
  book_id    SMALLINT      NOT NULL COMMENT '책 ID (FK)',
  chapter    SMALLINT           NOT NULL COMMENT '장',
  verse      SMALLINT           NOT NULL COMMENT '절',
  text       VARCHAR(2000) NOT NULL COMMENT '본문',
  reg_id     BIGINT        NOT NULL COMMENT '등록자 ID',
  reg_dt     TIMESTAMP     NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '등록일',
  upd_id     BIGINT        NULL COMMENT '수정자 ID',
  upd_dt     TIMESTAMP     NULL ON UPDATE CURRENT_TIMESTAMP  COMMENT '수정일', 
  PRIMARY KEY(verse_id), 
  UNIQUE KEY uq_bible_verse(version_id, book_id, chapter, verse)
) COMMENT '성경 본문';

  upd_dt        DATETIME      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP     COMMENT '수정일',
  PRIMARY KEY (meditation_id)
) COMMENT '묵상';
